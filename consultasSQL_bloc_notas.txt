#todos los valores de los inputs que son utilizados en las cosultas serán puestos como '%input%'

#asumimos que me dan el id de la región, no el nombre

SELECT P.nombre 
FROM Platos P, Restaurantes Rest, Region R
WHERE Rest.rid = R.id = '%input%'

2.- 
#asumimos que mostrar las habitaciones es mostrar el id de la habitación
SELECT Hab.habid, H.nombre
FROM Habitaciones Hab, Hoteles H
WHERE Hab.hid = H.hid AND H.estrellas > '%input%'
